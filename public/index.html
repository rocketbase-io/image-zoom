ZoomImage

Javascript component for partial zooming of images

npm i @rocketbase/zoom-image

Examples

ZoomImage.create("#zoom-image-1", {
  viewportClass: "shadow-xs shadow-sm shadow-md shadow-lg"
});
ZoomImage.create("#zoom-image-2", {
  factor: 4,
  scale: 200,
  shape: "square",
  position: "cursor",
  offset: [30, 30],
  viewportClass: "shadow-xs shadow-sm shadow-md shadow-lg bg-gray-200"
});
ZoomImage.create("#zoom-image-3", {
  factor: 2,
  position: "cover",
  zoomClass: "shadow-xs shadow-sm shadow-md shadow-lg",
  viewportClass: "shadow-inner bg-gray-200 rounded-lg"
});
ZoomImage.create("#zoom-image-4", {
  factor: 4,
  position: "fixed",
  placeAt: "#viewport",
  zoomClass: "shadow-xs shadow-sm shadow-md shadow-lg",
  viewportClass: "w-full h-full"
})
